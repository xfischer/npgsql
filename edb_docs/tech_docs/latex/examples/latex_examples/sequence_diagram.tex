\section*{Sequence Diagram with Bytefield Messages}

% Create two columns using minipage for manual control
\begin{minipage}{0.5\textwidth}
    % Left Column: Sequence Diagram
    \begin{sequencediagram}
        \newthread{actor1}{Actor 1}    % First actor/thread
        \newinst{actor2}{Actor 2}      % Second actor/instance

        % Actor 1 sends a message to Actor 2 (Message 1)
        \mess{actor1}{Message 1}{actor2}

        % Actor 2 responds to Actor 1 (Response)
        \mess{actor2}{Response}{actor1}

        % Actor 1 sends another message to Actor 2 (Message 2)
        \mess{actor1}{Message 2}{actor2}

        % Actor 2 sends a final message to Actor 1 (Final Message)
        \mess{actor2}{Final Message}{actor1}
    \end{sequencediagram}
\end{minipage}%
\begin{minipage}{0.5\textwidth}
    % Right Column: Bytefield Diagrams

    % Datagram for Message 1
    \noindent\textbf{Message 1:}
    \begin{bytefield}[bitwidth=0.6em]{16}

    \end{bytefield}

    \vspace{0.5cm} % Add some space between datagrams

    \end{bytefield}

    \vspace{0.5cm}

    % Datagram for Message 2

    \end{bytefield}

        \bitbox{8}{Type} & \bitbox{24}{Payload}
    \end{bytefield}
